MODULE ObjectAdr;

IMPORT M3ID, ObjectType, Tipe, Scope;

PROCEDURE Initialize () =
  VAR s: Scope.T;
  BEGIN
    s := Scope.PushNew (FALSE, M3ID.NoID);
    Scope.PopNew ();
    T := NIL;   (* the value is used by ObjectType.New ! *)
    T := ObjectType.New (NIL, FALSE, NIL, s, s);
    Tipe.Define ("_UNTRACED_ROOT", T, FALSE);
  END Initialize;

BEGIN
END ObjectAdr.
