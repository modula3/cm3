TYPE T <: REFANY;

PROCEDURE Now(): T RAISES {};

PROCEDURE Interval(t: T): T RAISES {};
Now() - 't'. 

PROCEDURE Add(t1, t2: T): T RAISES {};
't1' + 't2'. 

PROCEDURE AsString(t: T): TEXT RAISES {};
formatted as nn.mm. 

PROCEDURE Zero(): T RAISES {};
time or interval 00.00 

END M3Time.
